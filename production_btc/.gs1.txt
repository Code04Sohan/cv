/**
 * Verifies the incoming token matches the single active device session.
 */
function validateSession(incomingToken) {
  if (!incomingToken) return false;
  const activeToken = getProps().getProperty("SESSION_TOKEN");
  return (incomingToken === activeToken);
}

/**
 * Handles user login and enforces the Single-Session policy.
 */
function handleLogin(payload) {
  const props = getProps();
  let storedPass = props.getProperty("ADMIN_PASS");
  
  if (!storedPass) {
    // Graceful initialization fallback to "admin123" if script properties aren't set yet
    storedPass = "admin123";
    props.setProperty("ADMIN_PASS", storedPass);
  }

  if (payload.password === storedPass) {
    const newToken = Utilities.getUuid();
    props.setProperty("SESSION_TOKEN", newToken); // Invalidates old sessions
    return createResponse({ status: "success", token: newToken });
  }
  
  return createResponse({ status: "error", message: "Invalid Security Key." });
}

/**
 * ==========================================
 * 🚦 MODULE 3: ROUTING ENGINE (GATEKEEPER)
 * ==========================================
 */
function doPost(e) {
  try {
    let payload;
    if (e.postData && e.postData.contents) {
      try {
        payload = JSON.parse(e.postData.contents);
      } catch (err) {
        payload = e.parameter;
      }
    } else {
      payload = e.parameter;
    }
    
    if (!payload || !payload.action) {
        return createResponse({ status: "error", message: "Empty or invalid payload." });
    }
    
    const action = payload.action;

    // A. PUBLIC ROUTES (No auth required)
    if (action === "LOGIN") return handleLogin(payload);
    if (action === "REQUEST_RESET") return requestReset(payload);
    if (action === "COMPLETE_RESET") return verifyAndReset(payload);

    // B. THE GATEKEEPER - Enforce token checks on secure endpoints
    if (!validateSession(payload.token)) {
      return createResponse({ status: "error", message: "Unauthorized or Expired Session." });
    }

    // C. SECURE ROUTES
    switch (action) {
      case "VALIDATE_SESSION":
        return createResponse({ status: "success", message: "Session valid." });
      
      // Candidate Ingestion Engine integration
      case "ADD_CANDIDATE":
        return handleAddCandidate(payload.candidate);
        
      case "FETCH_DIRECTORY":
        return handleFetchDirectory(payload);
        
      case "EDIT_RECORD":
        return handleEditRecord(payload.record);

      case "GET_IMAGE_BASE64":
        return handleGetImageBase64(payload);

      case "DELETE_RECORD":
        return handleDeleteRecord(payload);

      // ====================================================
      // 💳 SHARDED PAYMENT LEDGER MODULE ROUTES (Logs_YYYY)
      // ====================================================
      case "GET_FEE_CONFIG":
        return handleGetFeeConfig();

      case "SET_FEE_CONFIG":
        return handleSetFeeConfig(payload.monthlyFee, payload.admissionFee);

      case "BULK_LOG_PAYMENTS":
        return handleBulkLogPayments(payload.payloadArray);

      case "LAZY_FETCH_STUDENT_LOGS":
        return handleLazyFetchStudentLogs(payload.studentId, payload.targetYear);

      case "FETCH_STATEMENT_RANGE":
        return handleFetchStatementRange(payload.startDate, payload.endDate);

      default:
        return createResponse({ status: "error", message: "Unknown action." });
    }
  } catch (error) {
    return createResponse({ status: "error", message: "Engine Error: " + error.toString() });
  }
}

/**
 * ==========================================
 * 🗑️ RECORD DELETION ENGINE
 * ==========================================
 */
function handleDeleteRecord(payload) {
  try {
    var spreadsheet = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID);
    var sheet = spreadsheet.getSheetByName(CONFIG.MASTER_SHEET_NAME);
    
    if (!sheet) {
      return ContentService.createTextOutput(JSON.stringify({ success: false, message: "Error: Database sheet not found." })).setMimeType(ContentService.MimeType.JSON);
    }

    var studentId = String(payload.STUDENT_ID);
    
    // Column Isolation Guard
    var idColumnRange = sheet.getRange(1, 1, sheet.getLastRow(), 1);
    var finder = idColumnRange.createTextFinder(studentId).matchEntireCell(true);
    var results = finder.findAll();

    // Validation Gate
    if (results.length === 0) {
      return ContentService.createTextOutput(JSON.stringify({ success: false, message: "Error: Student ID not found in database." })).setMimeType(ContentService.MimeType.JSON);
    }

    // Execution Block
    var targetRow = results[0].getRow();
    sheet.deleteRow(targetRow);

    // MIME Format Output
    var output = { success: true, message: "Record permanently deleted." };
    return ContentService.createTextOutput(JSON.stringify(output)).setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({ success: false, message: "Exception: " + error.toString() })).setMimeType(ContentService.MimeType.JSON);
  }
}

const OTP_EXPIRY_MS = 5 * 60 * 1000; // 5 minutes

function requestReset(payload) {
  const props = getProps();
  const otp = Math.floor(100000 + Math.random() * 900000).toString();
  
  props.setProperty("RESET_OTP", otp);
  props.setProperty("OTP_TIMESTAMP", new Date().getTime().toString());
  
  try {
    MailApp.sendEmail({
      to: CONFIG.ADMIN_EMAIL,
      subject: "Babla Yoga Training Center ERP Portal - Password Reset Required",
      body: `Your secure identity override verification token is: ${otp}\n\nThis token expires in 5 minutes.`
    });
    return createResponse({ status: "success", message: "OTP transmitted." });
  } catch (e) {
    return createResponse({ status: "error", message: "MailApp quota exceeded or unauthorized: " + e.toString() });
  }
}

function verifyAndReset(payload) {
  const props = getProps();
  const savedOtp = props.getProperty("RESET_OTP");
  const savedTimestamp = parseInt(props.getProperty("OTP_TIMESTAMP") || "0", 10);
  
  if (savedOtp && payload.otp === savedOtp && (new Date().getTime() - savedTimestamp) < OTP_EXPIRY_MS) {
    props.setProperty("ADMIN_PASS", payload.newPassword);
    props.deleteProperty("RESET_OTP");
    props.deleteProperty("OTP_TIMESTAMP");
    props.deleteProperty("SESSION_TOKEN"); // Terminate all other sessions
    return createResponse({ status: "success", message: "Security Key updated." });
  }
  return createResponse({ status: "error", message: "Invalid or expired OTP." });
}

/**
 * Converts a raw base64 string into a drive file, returning its public sharing URL.
 * Handles folder search and creation with multi-layered fallbacks.
 */
function processBase64File(fileName, mimeType, base64Data) {
  if (!base64Data) return "";
  
  const cleanData = base64Data.split(',').pop(); 
  const decoded = Utilities.base64Decode(cleanData);
  const blob = Utilities.newBlob(decoded, mimeType, fileName);
  
  // Auto-Healing Storage: get or create folder by name directly
  const parentFolder = getOrCreateDriveFolder("Babla Yoga Candidate Documents");
  
  const file = parentFolder.createFile(blob);
  
  // Set sharing properties viewable by anyone with the link
  file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
  
  return `https://drive.usercontent.google.com/download?id=${file.getId()}&export=view`;
}

function getOrCreateDriveFolder(folderName) {
  var folders = DriveApp.getFoldersByName(folderName);
  if (folders.hasNext()) {
    return folders.next();
  }
  return DriveApp.createFolder(folderName);
}

/**
 * Fetches all records from the directory securely mapped to headers
 */
function handleFetchDirectory(payload) {
  const spreadsheet = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID);
  const sheet = spreadsheet.getSheetByName(CONFIG.MASTER_SHEET_NAME);
  
  if (!sheet) {
    return createResponse({ status: "success", data: [] });
  }
  
  const dataRange = sheet.getDataRange();
  const allValues = dataRange.getValues();
  
  if (allValues.length <= 1) {
    return createResponse({ status: "success", data: [] });
  }
  
  const headers = allValues[0];
  const records = [];
  
  // Dynamically pack rows using the true runtime sheet headers
  for (let i = 1; i < allValues.length; i++) {
    const row = allValues[i];
    const record = {};
    for (let j = 0; j < headers.length; j++) {
      record[headers[j]] = row[j];
    }
    records.push(record);
  }
  
  return createResponse({ status: "success", data: records });
}

/**
 * Edits a specific record based on STUDENT_ID
 */
function handleEditRecord(record) {
  const spreadsheet = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID);
  const sheet = spreadsheet.getSheetByName(CONFIG.MASTER_SHEET_NAME);
  
  if (!sheet) {
    return createResponse({ status: "error", message: "Database not found." });
  }
  
  const dataRange = sheet.getDataRange();
  const allValues = dataRange.getValues();
  const headers = allValues[0];
  
  const studentIdIndex = headers.indexOf("STUDENT_ID");
  if (studentIdIndex === -1) {
    return createResponse({ status: "error", message: "Invalid schema mapping." });
  }
  
  let rowIndex = -1;
  for (let i = 1; i < allValues.length; i++) {
    if (String(allValues[i][studentIdIndex]) === String(record.STUDENT_ID)) {
      rowIndex = i + 1; // 1-indexed for spreadsheet, plus 1 for header
      break;
    }
  }
  
  if (rowIndex === -1) {
    return createResponse({ status: "error", message: "Student record not found." });
  }
  
  // Convert new base64 files if any
  if (record.STUDENT_PHOTO_URL && record.STUDENT_PHOTO_URL.startsWith("data:")) {
    record.STUDENT_PHOTO_URL = uploadCandidateBase64(record.STUDENT_PHOTO_URL, "STUDENT_" + record.STUDENT_NAME.replace(/\s+/g, "_") + "_" + record.STUDENT_ID);
  }
  
  if (record.STUDENT_SIGNATURE_URL && record.STUDENT_SIGNATURE_URL.startsWith("data:")) {
    record.STUDENT_SIGNATURE_URL = uploadCandidateBase64(record.STUDENT_SIGNATURE_URL, "SIGNATURE_" + record.STUDENT_NAME.replace(/\s+/g, "_") + "_" + record.STUDENT_ID);
  }
  
  const rowDataToSave = [];
  for (let j = 0; j < headers.length; j++) {
    const col = headers[j];
    // Keep old value if not provided in update, except when explicitly overwritten
    if (record[col] !== undefined) {
      rowDataToSave.push(record[col]);
    } else {
      rowDataToSave.push(allValues[rowIndex - 2][j]); 
    }
  }
  
  // Single bulk write back to the row
  sheet.getRange(rowIndex, 1, 1, headers.length).setValues([rowDataToSave]);
  
  return createResponse({ status: "success", message: "Record updated successfully." });
}

/**
 * ==========================================
 * 🖼️ MODULE 6B: SERVER-SIDE CORS IMAGE PROXY
 * ==========================================
 * Bypasses browser CORS entirely by fetching Google Drive file blobs
 * on the Apps Script server and returning them as Base64 data URIs.
 */

/**
 * Extracts a Google Drive File ID from any common Drive URL pattern,
 * fetches the file blob server-side, and returns a clean Base64 data URI.
 * Supports: /file/d/ID, open?id=ID, id=ID&export, uc?id=ID, download?id=ID
 */
function handleGetImageBase64(payload) {
  const url = payload.url;

  if (!url || typeof url !== 'string' || url.trim() === '') {
    return createResponse({ status: "error", message: "No URL provided." });
  }

  // If it's already a data URI, pass it straight back
  if (url.startsWith('data:')) {
    return createResponse({ status: "success", base64: url });
  }

  // Extract Google Drive File ID using multi-pattern regex matching
  var fileId = null;
  var patterns = [
    /\/d\/([a-zA-Z0-9_-]+)/,         // /file/d/FILE_ID/
    /[?&]id=([a-zA-Z0-9_-]+)/,        // ?id=FILE_ID or &id=FILE_ID
    /\/open\?id=([a-zA-Z0-9_-]+)/,    // /open?id=FILE_ID
    /uc\?.*id=([a-zA-Z0-9_-]+)/       // uc?export=download&id=FILE_ID
  ];

  for (var i = 0; i < patterns.length; i++) {
    var match = url.match(patterns[i]);
    if (match && match[1]) {
      fileId = match[1];
      break;
    }
  }

  if (!fileId) {
    return createResponse({ status: "error", message: "Could not extract Drive File ID from URL: " + url });
  }

  try {
    var file = DriveApp.getFileById(fileId);
    var blob = file.getBlob();
    var mimeType = blob.getContentType() || file.getMimeType();
    var bytes = blob.getBytes();
    var base64String = Utilities.base64Encode(bytes);
    var dataUri = "data:" + mimeType + ";base64," + base64String;

    return createResponse({ status: "success", base64: dataUri });
  } catch (err) {
    console.error("CORS Proxy fetch failed for fileId: " + fileId, err);
    return createResponse({ status: "error", message: "Drive file access error: " + err.toString() });
  }
}

/**
 * Parses files safely into Drive blobs and appends the fully populated candidate MasterRecord.
 */
function handleAddCandidate(candidate) {
  const spreadsheet = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID);
  let sheet = spreadsheet.getSheetByName(CONFIG.MASTER_SHEET_NAME);
  
  if (!sheet) {
    sheet = spreadsheet.insertSheet(CONFIG.MASTER_SHEET_NAME);
  }
  
  // Auto-Healing Database: Ensure headers exist if sheet is completely blank
  if (sheet.getLastRow() === 0 && sheet.getLastColumn() === 0) {
    const headers = SHEET_TEMPLATES.MasterRecord;
    sheet.appendRow(headers);
    sheet.setFrozenRows(1);
  }
  
  // ⚡ OPTIMIZED MEMORY READS: Pull database rows into RAM in a single partitioning call
  const dataRange = sheet.getDataRange();
  const allValues = dataRange.getValues();
  const headers = allValues[0];
  
  // Strict Multi-Column Duplicate Check
  if (allValues.length > 1) {
    const excludedFields = ['STUDENT_ID', 'TIMESTAMP', 'STUDENT_PHOTO_URL', 'STUDENT_SIGNATURE_URL'];
    const fieldsToCheck = headers.filter(h => !excludedFields.includes(h));
    
    const isDuplicate = allValues.slice(1).some(function(row) {
      return fieldsToCheck.every(function(field) {
        const fieldIndex = headers.indexOf(field);
        const rowVal = String(row[fieldIndex] || "").trim();
        const candidateVal = String(candidate[field] || "").trim();
        return rowVal === candidateVal;
      });
    });

    if (isDuplicate) {
      return createResponse({
        status: "error",
        message: "Duplicate student detected: An identical application with the same demographic and parental details already exists."
      });
    }
  }

  // 📂 FILE STREAMING: Convert base64 fields dynamically using our resilient processBase64File
  candidate.STUDENT_PHOTO_URL = uploadCandidateBase64(candidate.STUDENT_PHOTO_URL, "STUDENT_" + candidate.STUDENT_NAME.replace(/\s+/g, "_") + "_" + candidate.STUDENT_ID);
  candidate.STUDENT_SIGNATURE_URL = uploadCandidateBase64(candidate.STUDENT_SIGNATURE_URL, "SIGNATURE_" + candidate.STUDENT_NAME.replace(/\s+/g, "_") + "_" + candidate.STUDENT_ID);

  // Map candidate parameters to schema header template exactly
  const rowDataToSave = [];
  for (let i = 0; i < headers.length; i++) {
    const col = headers[i];
    if (col === "TIMESTAMP") {
      rowDataToSave.push(candidate.TIMESTAMP ? new Date(candidate.TIMESTAMP) : new Date());
    } else {
      rowDataToSave.push(candidate[col] !== undefined ? candidate[col] : "");
    }
  }
  
  // ⚡ HIGH-SPEED APPENDS: Single isolated write to Sheet
  sheet.appendRow(rowDataToSave);
  
  return createResponse({
    status: "success",
    message: `Candidate ${candidate.STUDENT_NAME} enrolled successfully!`,
    studentId: candidate.STUDENT_ID
  });
}

/**
 * Extracts mimetype and base64 fields, sending it to processBase64File
 */
function uploadCandidateBase64(base64Str, baseFilename) {
  if (!base64Str || !base64Str.startsWith("data:")) {
    return base64Str || ""; // Returns as-is if empty or already uploaded
  }
  
  try {
    const parts = base64Str.split(",");
    const mimeType = parts[0].match(/data:([^;]+);/)[1];
    const extension = mimeType.split("/")[1] || "png";
    const filename = baseFilename + "." + extension;
    
    return processBase64File(filename, mimeType, parts[1]);
  } catch (err) {
    console.error("Base64 candidate upload failed: " + baseFilename, err);
    return "Upload Error: " + err.toString();
  }
}

 * Tab naming convention: "Logs_" + YYYY (e.g., Logs_2025, Logs_2026)
 * Convention over Configuration: Tab names are NEVER stored in Script Properties.

/**
 * GET_FEE_CONFIG Handler
 * --------------------------------------------------
 * Reads the global fee rate configuration from PropertiesService.getScriptProperties().
 * Returns fallback defaults of 500 (monthly) and 1000 (admission) if the keys
 * have never been initialized, ensuring zero-config first-run behavior.
 *
 * Routing Trigger: case "GET_FEE_CONFIG" in doPost()
 *
 * Script Properties Keys:
 *   - MONTHLY_FEE: The per-month tuition fee charged to students
 *   - ADMISSION_FEE: The one-time enrollment fee charged at admission
 *
 * @returns {Object} { status: "success", monthlyFee: Number, admissionFee: Number }
 */
function handleGetFeeConfig() {
  try {
    var props = getProps();
    var rawMonthly = props.getProperty("MONTHLY_FEE");
    var rawAdmission = props.getProperty("ADMISSION_FEE");

    // Apply fallback defaults if keys are uninitialized (null)
    var monthlyFee = (rawMonthly !== null && rawMonthly !== "") ? Number(rawMonthly) : 500;
    var admissionFee = (rawAdmission !== null && rawAdmission !== "") ? Number(rawAdmission) : 1000;

    // Guard against NaN from corrupted property values
    if (isNaN(monthlyFee)) monthlyFee = 500;
    if (isNaN(admissionFee)) admissionFee = 1000;

    return createResponse({
      status: "success",
      monthlyFee: monthlyFee,
      admissionFee: admissionFee
    });
  } catch (error) {
    return createResponse({
      status: "error",
      message: "Fee Config Read Error: " + error.toString()
    });
  }
}

/**
 * SET_FEE_CONFIG Handler
 * --------------------------------------------------
 * Casts incoming parameters to clean Numbers, validates them,
 * writes them to the Cloud Script Properties keys (MONTHLY_FEE, ADMISSION_FEE),
 * and returns a success confirmation token.
 *
 * Routing Trigger: case "SET_FEE_CONFIG" in doPost()
 *
 * @param {*} monthlyFee - The new monthly fee value (will be cast to Number)
 * @param {*} admissionFee - The new admission fee value (will be cast to Number)
 * @returns {Object} { status: "success", message: "...", monthlyFee: Number, admissionFee: Number }
 */
function handleSetFeeConfig(monthlyFee, admissionFee) {
  try {
    // Cast to clean Numbers — strips strings, booleans, etc.
    var cleanMonthly = Number(monthlyFee);
    var cleanAdmission = Number(admissionFee);

    // Validation Gate: Reject NaN or negative values
    if (isNaN(cleanMonthly) || cleanMonthly < 0) {
      return createResponse({
        status: "error",
        message: "Invalid monthly fee value. Must be a non-negative number."
      });
    }
    if (isNaN(cleanAdmission) || cleanAdmission < 0) {
      return createResponse({
        status: "error",
        message: "Invalid admission fee value. Must be a non-negative number."
      });
    }

    // Persist to Cloud Script Properties
    var props = getProps();
    props.setProperty("MONTHLY_FEE", String(cleanMonthly));
    props.setProperty("ADMISSION_FEE", String(cleanAdmission));

    return createResponse({
      status: "success",
      message: "Fee configuration updated successfully.",
      monthlyFee: cleanMonthly,
      admissionFee: cleanAdmission
    });
  } catch (error) {
    return createResponse({
      status: "error",
      message: "Fee Config Write Error: " + error.toString()
    });
  }
}

/**
 * BULK_LOG_PAYMENTS Handler (Sharded Transactional Bulk-Writer)
 * --------------------------------------------------
 * Accepts an incoming array of month payment objects, dynamically resolves
 * the target annual shard tab name as "Logs_" + transactionYear, auto-creates
 * the tab with schema headers if it doesn't exist, then injects all rows
 * simultaneously using a single high-speed .setValues() call to bypass
 * timeout gates.
 *
 * Routing Trigger: case "BULK_LOG_PAYMENTS" in doPost()
 *
 * @param {Array<Object>} payloadArray - Array of payment objects, each containing:
 *   {
 *     TXN_ID: string,          // Unique transaction trace token (e.g. "TXN-2026-A3F7B")
 *     TIMESTAMP: string,       // ISO date string of payment time
 *     STUDENT_ID: string,      // Linked student identifier
 *     RL_NO: string,           // Roll number reference
 *     STUDENT_NAME: string,    // Full candidate name
 *     FEE_PERIOD: string,      // Month-Year label (e.g. "January-2026")
 *     STATUS: string,          // Payment status (e.g. "PAID")
 *     AMOUNT_COLLECTED: number // Amount in INR
 *   }
 *
 * Returns: { status: "success", success: true, message: "...", rowsWritten: N }
 */
function handleBulkLogPayments(payloadArray) {
  try {
    // Input Validation Gate
    if (!payloadArray || !Array.isArray(payloadArray) || payloadArray.length === 0) {
      return createResponse({
        status: "error",
        success: false,
        message: "Invalid payload: Expected a non-empty array of payment objects."
      });
    }

    var ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID);
    var schema = SHEET_TEMPLATES.PaymentLog;

    // -------------------------------------------------------
    // PHASE A: Parse the transaction year from FEE_PERIOD.
    // Handles both "2026-06" (New Standard) and "June-2026" (Legacy)
    // -------------------------------------------------------
    var transactionYear = null;
    var firstEntry = payloadArray[0];

    if (firstEntry.FEE_PERIOD && typeof firstEntry.FEE_PERIOD === 'string') {
      var periodParts = firstEntry.FEE_PERIOD.split("-");
      if (periodParts.length >= 2) {
        // First try the new standard (YYYY-MM)
        var parsedYear = parseInt(periodParts[0], 10);
        
        // If it fails, try the legacy standard (Month-YYYY)
        if (isNaN(parsedYear) || parsedYear < 2000) {
            parsedYear = parseInt(periodParts[periodParts.length - 1], 10);
        }

        if (!isNaN(parsedYear) && parsedYear >= 2000 && parsedYear <= 2099) {
          transactionYear = parsedYear;
        }
      }
    }

    // Fallback: derive year from TIMESTAMP
    if (transactionYear === null && firstEntry.TIMESTAMP) {
      var tsDate = new Date(firstEntry.TIMESTAMP);
      if (!isNaN(tsDate.getTime())) {
        transactionYear = tsDate.getFullYear();
      }
    }

    // Ultimate fallback: current server year
    if (transactionYear === null) {
      transactionYear = new Date().getFullYear();
    }

    // -------------------------------------------------------
    // PHASE B: Resolve target shard sheet — Convention over Configuration
    // -------------------------------------------------------
    var sheetName = "Logs_" + transactionYear;
    var sheet = ss.getSheetByName(sheetName);

    // Auto-Creation Guard: If shard tab doesn't exist, create it with schema headers
    if (!sheet) {
      sheet = ss.insertSheet(sheetName);
      sheet.appendRow(schema);
      sheet.setFrozenRows(1);
      sheet.getRange(1, 1, 1, schema.length).setFontWeight("bold");
    }

    // Auto-Healing: If shard exists but is completely empty, re-inject headers
    if (sheet.getLastRow() === 0) {
      sheet.appendRow(schema);
      sheet.setFrozenRows(1);
      sheet.getRange(1, 1, 1, schema.length).setFontWeight("bold");
    }

    // -------------------------------------------------------
    // PHASE C: Build the 2D matrix aligned to schema column order
    // -------------------------------------------------------
    var matrix = [];
    for (var i = 0; i < payloadArray.length; i++) {
      var entry = payloadArray[i];
      var row = [];
      for (var j = 0; j < schema.length; j++) {
        var key = schema[j];
        if (key === 'TIMESTAMP' && entry[key]) {
          // Ensure timestamps are stored as proper Date objects for sheet formatting
          row.push(new Date(entry[key]));
        } else {
          row.push(entry[key] !== undefined && entry[key] !== null ? entry[key] : "");
        }
      }
      matrix.push(row);
    }

    // -------------------------------------------------------
    // PHASE D: ⚡ Atomic Batch Write — Single execution boundary
    // -------------------------------------------------------
    var lastRow = sheet.getLastRow();
    sheet.getRange(lastRow + 1, 1, matrix.length, schema.length).setValues(matrix);
    
    return createResponse({
      status: "success",
      success: true,
      message: "Successfully logged " + matrix.length + " payment transaction(s) to " + sheetName + ".",
      rowsWritten: matrix.length,
      targetSheet: sheetName
    });
    
  } catch (error) {
    return createResponse({
      status: "error",
      success: false,
      message: "Bulk Payment Write Error: " + error.toString()
    });
  }
}

/**
 * LAZY_FETCH_STUDENT_LOGS Handler (Targeted Single-Candidate Log Fetcher)
 * --------------------------------------------------
 * Fetches payment log records for a single specific student from the annual
 * shard tab matching the targetYear parameter. Uses a precise column-restricted
 * TextFinder search on Column C (STUDENT_ID) with matchEntireCell(true) to
 * extract only that student's payment rows for the given year.
 *
 * This is designed for lazy backward-chaining waterfall lookups — the frontend
 * calls this API once per year, walking backwards through time to build a
 * continuous payment timeline until it hits the admission month.
 *
 * Routing Trigger: case "LAZY_FETCH_STUDENT_LOGS" in doPost()
 *
 * @param {string} studentId - The STUDENT_ID to search for in Column C
 * @param {number|string} targetYear - The year to target (e.g., 2026, "2025")
 * @returns {Object} { status: "success", data: Array<Object>, targetYear: Number, sheetName: String }
 *   Each object in data matches: { TXN_ID, TIMESTAMP, STUDENT_ID, RL_NO, STUDENT_NAME, FEE_PERIOD, STATUS, AMOUNT_COLLECTED }
 */
function handleLazyFetchStudentLogs(studentId, targetYear) {
  try {
    // Input Validation
    if (!studentId || studentId === "") {
      return createResponse({
        status: "error",
        message: "Missing required parameter: studentId."
      });
    }

    var year = parseInt(targetYear, 10);
    if (isNaN(year) || year < 2000 || year > 2099) {
      return createResponse({
        status: "error",
        message: "Invalid targetYear. Must be a 4-digit year between 2000 and 2099."
      });
    }

    var ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID);
    var sheetName = "Logs_" + year;
    var sheet = ss.getSheetByName(sheetName);

    // If the shard tab doesn't exist for this year, return empty results
    // (student has no payment records for this year — perfectly valid)
    if (!sheet) {
      return createResponse({
        status: "success",
        data: [],
        targetYear: year,
        sheetName: sheetName,
        message: "No payment log sheet found for year " + year + ". Sheet '" + sheetName + "' does not exist."
      });
    }

    var lastRow = sheet.getLastRow();

    // If only headers exist (or sheet is empty), return empty dataset
    if (lastRow <= 1) {
      return createResponse({
        status: "success",
        data: [],
        targetYear: year,
        sheetName: sheetName,
        message: "Payment log sheet for " + year + " exists but contains no data rows."
      });
    }

    var schema = SHEET_TEMPLATES.PaymentLog;

    // -------------------------------------------------------
    // COLUMN-RESTRICTED TEXTFINDER SEARCH
    // Column C (index 3 in 1-based) = STUDENT_ID per schema definition:
    //   ['TXN_ID'(1), 'TIMESTAMP'(2), 'STUDENT_ID'(3), ...]
    // Use createTextFinder on Column C only for precise isolation.
    // -------------------------------------------------------
    var studentIdColumn = sheet.getRange(2, 3, lastRow - 1, 1); // Column C, rows 2 to lastRow
    var finder = studentIdColumn.createTextFinder(String(studentId)).matchEntireCell(true);
    var matchedCells = finder.findAll();

    // If no matches found, return empty array (student has no records this year)
    if (!matchedCells || matchedCells.length === 0) {
      return createResponse({
        status: "success",
        data: [],
        targetYear: year,
        sheetName: sheetName,
        message: "No payment records found for student '" + studentId + "' in " + sheetName + "."
      });
    }

    // -------------------------------------------------------
    // EXTRACT MATCHED ROWS — Map each found row to schema objects
    // -------------------------------------------------------
    var results = [];
    for (var i = 0; i < matchedCells.length; i++) {
      var matchedCell = matchedCells[i];
      var rowNumber = matchedCell.getRow();

      // Read the full row data for this match
      var rowData = sheet.getRange(rowNumber, 1, 1, schema.length).getValues()[0];

      var record = {};
      for (var j = 0; j < schema.length; j++) {
        var key = schema[j];
        var cellValue = rowData[j];

        // Coerce Date objects to ISO strings for clean frontend parsing
        if (key === 'TIMESTAMP' && cellValue instanceof Date) {
          record[key] = cellValue.toISOString();
        } else {
          record[key] = cellValue;
        }
      }
      results.push(record);
    }

    return createResponse({
      status: "success",
      data: results,
      targetYear: year,
      sheetName: sheetName,
      message: "Found " + results.length + " payment record(s) for student '" + studentId + "' in " + sheetName + "."
    });
  } catch (error) {
    return createResponse({
      status: "error",
      message: "Lazy Fetch Student Logs Error: " + error.toString()
    });
  }
}

/**
 * FETCH_STATEMENT_RANGE Handler
 * --------------------------------------------------
 * Parametric Range Aggregator Backend
 * Dynamically slices cross-year historical logs.
 */
function handleFetchStatementRange(startDateStr, endDateStr) {
  try {
    var ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID);
    var start = new Date(startDateStr);
    var end = new Date(endDateStr);
    
    if (isNaN(start.getTime()) || isNaN(end.getTime())) {
      return createResponse({ status: 'error', message: 'Invalid date parameters.' });
    }
    
    // Ensure end date covers the full day
    end.setHours(23, 59, 59, 999);
    
    var startYear = start.getFullYear();
    var endYear = end.getFullYear();
    
    var mergedLogs = [];
    var headers = SHEET_TEMPLATES.PaymentLog;
    
    for (var y = startYear; y <= endYear; y++) {
      var sheetName = 'Logs_' + y;
      var sheet = ss.getSheetByName(sheetName);
      if (!sheet) continue;
      
      var lastRow = sheet.getLastRow();
      if (lastRow < 2) continue;
      
      var data = sheet.getRange(2, 1, lastRow - 1, headers.length).getValues();
      
      for (var i = 0; i < data.length; i++) {
        var row = data[i];
        var txnDate = new Date(row[1]); // TIMESTAMP is column 1 (0-indexed array)
        
        if (txnDate >= start && txnDate <= end) {
          var logObj = {};
          for (var j = 0; j < headers.length; j++) {
            logObj[headers[j]] = row[j];
          }
          mergedLogs.push(logObj);
        }
      }
    }
    
    return createResponse({ status: 'success', data: mergedLogs });
  } catch (error) {
    return createResponse({ status: 'error', message: 'Fetch Range Error: ' + error.toString() });
  }
}
